class Variational_AE(nn.Module):
  def __init__(self,input_unit,hidden_unit1,hidden_unit2,k):
    self.input_unit=input_unit
    self.hidden_unit1=hidden_unit1
    self.hidden_unit2=hidden_unit2
    self.k=k
    super(Variational_AE,self).__init__()
    
    self.l1=nn.Linear(input_unit,hidden_unit1)
    self.l2=nn.Linear(hidden_unit1,hidden_unit2)
    self.l3=nn.Linear(hidden_unit2,k)
    self.l4=nn.Linear(hidden_unit2,k)
    

    self.l5=nn.Linear(k,hidden_unit2)
    self.l6=nn.Linear(hidden_unit2, hidden_unit1)
    self.l7=nn.Linear(hidden_unit1, input_unit)


    self.sigmoid=nn.Sigmoid()
    self.relu=nn.ReLU()

  def reparameter(self,me,sig):
    si=sig.mul(0.5).exp_()
    ep=torch.randn(*me.size())
    return me.cuda()+si.cuda()*ep.cuda()

  def encoding(self,inp):
    o1=self.relu(self.l1(inp))
    o2=self.relu(self.l2(o1))
    o3=self.l3(o2)
    o4=self.l4(o2)
    return o3,o4
  
  def decoding(self,dec_inp):
    o1=self.relu(self.l5(dec_inp))
    o2=self.relu(self.l6(o1))
    sig_out=self.sigmoid(self.l7(o2))
    return sig_out
  
  def forward(self,inp):
    inp=inp.view(-1,self.input_unit)
    me,sig=self.encoding(inp)
    dec_inp=self.reparameter(me,sig)
    dec_out=self.decoding(dec_inp)
    return dec_out,me,sig
